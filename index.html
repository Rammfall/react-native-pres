Structure of files for screens or components

index.js // Component with structure(naming ComponentScreen if it screen)
styled.js // Contains all styled components for component or screen

If you support both platforms

index.js // Component with structure(naming ComponentScreen if it screen)
styled.android.js // Contains all styled components for component or screen for android
styled.ios.js // Contains all styled components for component or screen for ios

In garage project right way to import styled components will be:

import * as S from './styled';
const Component = ({children}) => (
<S.Wrapper>{children}</S.Wrapper>
);

If we have dynamic properties

import * as S from './styled';
const Component = ({children, style}) => (
<S.Wrapper style={style}>{children}</S.Wrapper>
);

In styled

export const Wrapper = styled(Component)`
margin-bottom: 22px;
`;

Work with IcoMoon

export const CustomIcon = styled(IcoMoonIcon).attrs((props) => ({
name: 'seach',
color: props.theme.colors.black,
size: 24,
}))`
margin-bottom: 22px;
`;

For modals(with animation) use react-native-modal

import Modal from 'react-native-modal';
export const CustomModal = styled(Modal)`
margin-bottom: 22px;
`;

I recommend remove margins and paddings from inputs

Because it have very big difference on ios and android

For images use react-native-fast-image

import FastImage from 'react-native-fast-image';
export const CustomImage = styled(FastImage)`
margin-bottom: 22px;
`;

For platform specific part of component

import {Platform} from 'react-native';
console.log(Platgorm.OS) // return ios or android

React native analogue calc function

import {Dimensions} from 'react-native';
console.log(Dimensions.get('window').height) // return height
console.log(Dimensions.get('window').width) // return width
